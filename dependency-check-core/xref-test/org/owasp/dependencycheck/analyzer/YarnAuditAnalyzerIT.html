19      @Test
20      public void testAnalyzePackageYarn() throws AnalysisException, InitializationException, InvalidSettingException {
21          
22          //Assume.assumeThat(getSettings().getBoolean(Settings.KEYS.ANALYZER_YARN_AUDIT_ENABLED), is(true));
23          try (Engine engine = new Engine(getSettings())) {
24              YarnAuditAnalyzer analyzer = new YarnAuditAnalyzer();
25              analyzer.setFilesMatched(true);
26              analyzer.initialize(getSettings());
27              analyzer.prepare(engine);
28              final Dependency toScan = new Dependency(BaseTest.getResourceAsFile(this, "yarnaudit/yarn.lock"));
29              analyzer.analyze(toScan, engine);
30              boolean found = false;
31              assertTrue("More then 1 dependency should be identified", 1 < engine.getDependencies().length);
32              for (Dependency result : engine.getDependencies()) {
33                  if ("yarn.lock?uglify-js".equals(result.getFileName())) {
34                      found = true;
35                      assertTrue(result.getEvidence(EvidenceType.VENDOR).toString().contains("uglify-js"));
36                      assertTrue(result.getEvidence(EvidenceType.PRODUCT).toString().contains("uglify-js"));
37                      assertTrue(result.getEvidence(EvidenceType.VERSION).toString().contains("3.12.4"));
38                      assertTrue(result.isVirtual());
39                  }
40              }
41              assertTrue("Uglify was not found", found);
42          } catch (InitializationException ex) {
43              //yarn is not installed - skip the test case.
44              Assume.assumeNoException(ex);
45          }
46      }
