1   package org.owasp.dependencycheck.utils;

3   import java.util.Objects;
4   
5   /**
6    * Default implementation of a {@code CveUrlParser}.
7    *
8    * @author nhumblot
9    *
10   */
11  public final class DefaultCveUrlModifiedParser implements CveUrlParser {
12  
13    private static final String URL_SEPARATOR = "/";
14  
15    private final Settings settings;
16  
17    DefaultCveUrlModifiedParser(Settings settings) {
18      this.settings = settings;
19    }
20  
21    @Override
22    public String getDefaultCveUrlModified(String baseUrl) {
23      String defaultBaseUrlEnd = URL_SEPARATOR + settings.getString(Settings.KEYS.CVE_BASE_DEFAULT_FILENAME);
24      if (Objects.nonNull(baseUrl) && baseUrl.endsWith(defaultBaseUrlEnd)) {
25        String defaultModifiedUrlEnd = URL_SEPARATOR + settings.getString(Settings.KEYS.CVE_MODIFIED_DEFAULT_FILENAME);
26        return baseUrl.substring(0, baseUrl.length() - defaultBaseUrlEnd.length()) + defaultModifiedUrlEnd;
27      }
28      return null;
29    }
30  }
